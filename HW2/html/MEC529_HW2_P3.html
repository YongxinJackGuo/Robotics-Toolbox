%-----question (a)----------------
function R = AxisAngle_to_Rot(axis,angle)
% assign the variables for convenience
w1 = axis(1);
w2 = axis(2);
w3 = axis(3);
% create a template for the skew-symmetric matrix.
skew = zeros(3,3);
% create the skew-symmetric matrix
skew(2,1) = w3;
skew(1,2) = -1*skew(2,1);
skew(1,3) = w2;
skew(3,1) = -1*skew(1,3);
skew(3,2) = w1;
skew(2,3) = -1*skew(3,2);
% assign cosine and sine to variables
C = cos(angle);
S = sin(angle);
v = 1-C;
% compute the Rotation matrix using Equation (2.4) in lecture notes.
R = eye(3)+skew*S+skew*skew*v;
end
%-----question (a) ends------------


%-----question (b)-----------------
function [axis,angle] = Rot_to_AxisAngle(R)
% compute the trace of R
Trace = R(1,1)+R(2,2)+R(3,3);
% compute the angle of rotation
angle = acos((Trace-1)/2);
%compute the axis of rotation
axis = (1/(2*sin(angle)))*[R(3,2)-R(2,3); R(1,3)-R(3,1); R(2,1)-R(1,2)];
end
